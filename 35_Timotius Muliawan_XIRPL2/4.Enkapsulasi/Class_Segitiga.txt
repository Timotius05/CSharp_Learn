        private int alas;
        private int tinggi;

        }

            return tinggi;
        }
        public int settinggi(int ntinggi)
        {
            tinggi += ntinggi;
            return ntinggi;
        }
        public int getsisim()
        {
            return sisim;
        }
        public int setsisim(int nsisim)
        {
            sisim += nsisim;
            return nsisim;
